       }

    2. In `src/db/cooking-history.ts`, add two new exported functions:

       `getCookedRecipesWithMeta(db: SQLiteDatabase): Promise<CookedRecipeMeta[]>` -- uses SQL:
       ```sql
       SELECT
         recipe_id,
         COUNT(*) as cook_count,
         (SELECT rating FROM cooking_history ch2
          WHERE ch2.recipe_id = ch.recipe_id
          ORDER BY cooked_at DESC LIMIT 1) as latest_rating,
         MAX(cooked_at) as last_cooked_at
       FROM cooking_history ch
       GROUP BY recipe_id
       ORDER BY MAX(cooked_at) DESC
       ```
       Maps snake_case DB columns to camelCase CookedRecipeMeta fields.

       `updateLatestRating(db: SQLiteDatabase, recipeId: string, rating: number): Promise<void>` -- uses SQL:
       ```sql
       UPDATE cooking_history SET rating = ?
       WHERE id = (
         SELECT id FROM cooking_history
         WHERE recipe_id = ?
         ORDER BY cooked_at DESC LIMIT 1
       )
       ```
