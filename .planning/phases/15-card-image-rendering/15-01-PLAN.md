**Update build-images.ts:**

1. Add `import { encode } from 'blurhash';` at the top of the file.

2. Add a `generateBlurhash` helper function:
```typescript
async function generateBlurhash(imagePath: string): Promise<string> {
  const { data, info } = await sharp(imagePath)
    .raw()
    .ensureAlpha()
    .resize(32, 32, { fit: 'inside' })
    .toBuffer({ resolveWithObject: true });
  return encode(
    new Uint8ClampedArray(data),
    info.width,
    info.height,
    4,  // componentX
    3,  // componentY
  );
}
```

3. Update the `processedImages` Map type to include blurhash fields:
```typescript
const processedImages: Map<

```
