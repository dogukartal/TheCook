  <action>
    Replace TheCook/app/_layout.tsx with the integrated version following Pattern 1 from 02-RESEARCH.md:

    Key changes from current layout:
    1. Wrap the Stack in `SessionProvider`
    2. Add a child component (e.g., `RootNavigator`) that reads the `onboarding_completed` flag from SQLite and redirects to onboarding if needed
    3. Call `initAuthListener(db)` inside the `SQLiteProvider.onInit` callback (after migrateDb + seedIfNeeded)
    4. Register onboarding, (auth), and modal Stack.Screen entries in addition to existing (tabs) and modal

    Pitfall 5 warning: Do NOT redirect before the root layout mounts. Use a loading state:

    ```typescript
    import { Stack, Redirect } from 'expo-router';
    import { SQLiteProvider, useSQLiteContext } from 'expo-sqlite';
    import { useState, useEffect } from 'react';
    import { migrateDb } from '../src/db/client';
    import { seedIfNeeded } from '../src/db/seed';
    import { initAuthListener } from '../src/auth/sync';
    import { SessionProvider } from '../src/auth/useSession';
    import { useProfileDb } from '../src/db/profile';

    function RootNavigator() {
      const db = useSQLiteContext();
      const { getProfile } = useProfileDb();
      const [onboardingDone, setOnboardingDone] = useState<boolean | null>(null);

      useEffect(() => {
        getProfile().then(p => setOnboardingDone(p.onboardingCompleted));
      }, []);

      if (onboardingDone === null) return null; // splash / loading — do not redirect yet

      return (
        <Stack screenOptions={{ headerShown: false }}>
          <Stack.Screen name="(tabs)" />
          <Stack.Screen name="onboarding" />
          <Stack.Screen name="(auth)" />
          <Stack.Screen name="modal" options={{ presentation: 'modal' }} />
          {!onboardingDone && <Redirect href="/onboarding/allergens" />}
        </Stack>
      );
    }

    export default function RootLayout() {
      return (
        <SQLiteProvider
          databaseName="thecook.db"
          onInit={async (db) => {
            await migrateDb(db);
            await seedIfNeeded(db);
            initAuthListener(db); // start auth listener after DB is ready
          }}
        >
          <SessionProvider>
            <RootNavigator />
          </SessionProvider>
        </SQLiteProvider>
      );
    }
    ```

    Note: `<Redirect>` inside Stack renders only if condition is true. Place it after the Screen declarations. Verify this works with Expo Router v6 — if Redirect inside Stack causes issues, use `router.replace('/onboarding/allergens')` inside the useEffect instead, but only after `onboardingDone === false` is confirmed (never redirect while onboardingDone is null).
